full = subsetLoops(full, full@rowData$loopWidth >= 5000)
full = removeSelfLoops(full)

Now lets filter the loops that have >= 5 PETs in one Rep but 0 in the other

cm <- full_corrected@counts
EBNA3A_Off <- ((cm[,2]>=5 &cm[,3]==0)|(cm[,3]>=5&cm[,2]==0))
EBNA3A_On <- ((cm[,5]>=5 &cm[,6]==0)|(cm[,6]>=5&cm[,5]==0))
qc_filt <- subsetLoops(full_corrected, !(EBNA3A_Off | EBNA3A_On))

enhancerbed = "wgEncodeBroadHmmGm12878HMMhg19_Enhancers.bed"
ctcfbed = "CTCFhuman.Rep.1.reg_peaks.narrowPeak"
enhancer = rmchr(padGRanges(bedToGRanges(enhancerbed), pad = 2000))

# Annotate the qcLoops object
